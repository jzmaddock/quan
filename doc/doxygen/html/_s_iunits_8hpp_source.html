00025 const int maxPowerTen = 38;  //
00026 // == std::numeric_limits<float>::max_exponent10;
00027 // == FLT_MAX_10_EXP in float.h
00028 // so can represent FLT_MAX = 3.402823466e+38F;
00029 // down to  FLT_MIN_10_EXP in float.h
00030 // so can represent FLT_MIN = 1.175494351e–38F;
00031 // (Done to limit the size of the power of Ten lookup table below,
00032 // and make code same if store value in a float.
